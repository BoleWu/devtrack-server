三、统一返回体（企业必备）
common/result/R.java
@Data
@AllArgsConstructor
@NoArgsConstructor
public class R<T> {
private int code;
private String msg;
private T data;

    public static <T> R<T> ok(T data){
        return new R<>(200, "success", data);
    }

    public static R<?> fail(String msg){
        return new R<>(500, msg, null);
    }
}
